    private void btnSaveActionPerformed(java.awt.event.ActionEvent evt) {                                        
        try {


            //if record exists for this month, display message
            //else save record at end of file
            FileReader reader = new FileReader("test.txt");
            BufferedReader br = new BufferedReader(reader);
            String actualReadLine;
            boolean flag = false;

            //read from file and compare month
            //if match found, then display message
            while ((actualReadLine = br.readLine()) != null) {
                if (actualReadLine.equals(ComboBoxMonth.getSelectedItem().toString())) {
                    flag = true;
                }
            }
            br.close();
            if (flag) {
                JOptionPane.showMessageDialog(null, "Record already exists for this month!");
            } else {
                FileWriter writer = new FileWriter("test.txt", true);//set to true so that we append data at the end of the file
                BufferedWriter bw = new BufferedWriter(writer);
                bw.write(ComboBoxMonth.getSelectedItem().toString());
                bw.newLine();
                bw.write(txtFood.getText());
                bw.newLine();
                bw.write(txtBills.getText());
                bw.newLine();
                bw.write(txtOthers.getText());
                bw.newLine();
                bw.write(txtTotal.getText());
                bw.newLine();
                bw.close();
                JOptionPane.showMessageDialog(null, "Record saved successfully!");
                txtFood.setText("");
                txtBills.setText("");
                txtOthers.setText("");
                txtTotal.setText("");
            }
        } catch (IOException e) {
            JOptionPane.showMessageDialog(null, e.getMessage());
        }

    }                                       

    private void btnSearchActionPerformed(java.awt.event.ActionEvent evt) {                                          
        String month = ComboBoxMonth.getSelectedItem().toString();
        try {
            FileReader reader = new FileReader("test.txt");
            BufferedReader br = new BufferedReader(reader);
            String actualReadLine;
            boolean flag = false;

            //read from file and compare month
            //if match found, then read expenses and display, otherwise move to next record
            while ((actualReadLine = br.readLine()) != null) {
                if (actualReadLine.equals(month)) {
                    txtFood.setText(br.readLine());
                    txtBills.setText(br.readLine());
                    txtOthers.setText(br.readLine());
                    txtTotal.setText(br.readLine());
                    flag = true;
                }
            }
            if (!flag) {
                JOptionPane.showMessageDialog(null, "Sorry, no record found. Try for another month!");
            }
            br.close();
        } catch (IOException e) {
            JOptionPane.showMessageDialog(null, e.getMessage());
        }

    }                                         

    private void btnTotalActionPerformed(java.awt.event.ActionEvent evt) {                                         
        try {

            Double num1, num2, num3, total;

            num1 = Double.parseDouble(txtFood.getText());
            num2 = Double.parseDouble(txtBills.getText());
            num3 = Double.parseDouble(txtOthers.getText());
            total = num1 + num2 + num3;

            txtTotal.setText("" + total);

        } catch (NumberFormatException e) {
            JOptionPane.showMessageDialog(null, "Check numbers again.");

        }     // Calculates and sums all the values that user inputs. However if user does not input value, an error message is shown
    }
